# Linux系统依赖安装说明（Python 3.13.1兼容）：
# 
# Ubuntu/Debian/deepin/UOS:
# sudo apt update
# sudo apt install -y python3.13-dev python3.13-venv python3.13-pip
# sudo apt install -y pulseaudio pulseaudio-utils alsa-utils espeak libasound2-dev libpulse-dev libxcb-cursor-dev libxcb-xinerama0 libxcb-shape0 libxcb-keysyms1-dev libxcb-image0-dev libxcb-render-util0-dev libxkbcommon-dev libxcb-cursor0
# 
# Fedora/RHEL/CentOS:
# sudo dnf update
# sudo dnf install -y python3.13 python3.13-devel python3.13-pip
# sudo dnf install -y pulseaudio pulseaudio-utils alsa-utils espeak alsa-lib-devel pulseaudio-libs-devel libxcb-cursor-devel libxcb-keysyms-devel libxcb-image-devel libxcb-render-util-devel libxkbcommon-devel
# 
# Arch Linux/Manjaro:
# sudo pacman -Syu
# sudo pacman -S python3.13 python3.13-pip
# sudo pacman -S pulseaudio pulseaudio-alsa alsa-utils espeak alsa-lib pulseaudio libxcb-cursor libxcb-keysyms libxcb-image libxcb-render-util libxkbcommon
# 
# openSUSE:
# sudo zypper refresh
# sudo zypper install -y python3.13 python3.13-devel python3.13-pip
# sudo zypper install -y pulseaudio pulseaudio-utils alsa-utils espeak alsa-devel pulseaudio-devel libxcb-cursor0 libxcb-keysyms1 libxcb-image0 libxcb-render-util0 libxkbcommon-devel
# 
# 音频功能需要用户在audio组中：
